React Form Data Handling

to handle a data of input of form using react, we have have make input component as Controlled Component

Controlled Component

Exmaple: <input value = {this.state.name} onChange={this.changeName}/>
In this example value of input handles by Component and onChange also handle by Component means every input value is controlled by react component then this input field called as Controlled Component

Example

        class FormC extends Component {
            constructor(props) {
                super(props)
                this.state = {
                    name: "",
                    password: ""
                }
                this.changeName = this.changeName.bind(this)
            }
            changeName(event){
                this.setState(prevState=>({
                    name: event.target.value
                }))
            }
            changePassword = (event)=>{
                this.setState(prevState=>({
                    password: event.target.value
                }))
            }
        
            // handling the form submission
            validateHandler=(event)=>{
                if(this.state.name.length>6 && this.state.password.length>=6){
                    alert(`valid input ${this.state.name} ${this.state.password}`)
                }else{
                    alert("invalid")
                }
                // to prevent the refresh after the submission
                event.preventDefault();
            }
            render() {
                return (
                    <div>
                        <form onSubmit={this.validateHandler}>
                            <div>
                            <label>Name: </label>
                            <input type='text' value={this.state.name} onChange={this.changeName}/><p></p>
                            </div>
                            <div>
                                <label>Password: </label>
                                <input type="password" value={this.state.password} onChange={this.changePassword}/><p></p>
                            </div>
                            <button type='submit'>Submit</button>
                        </form>
                    </div>
                )
            }
        }
