Child To Parent Component - method as props

To change something in Parent Component using Child Component - this is done by passing a function as props of child Component

Step-by-Step

1. define a function in parent

2. pass this function to child as props

3. in child call the function using props


example:

        Parent Component
        class Parent extends Component {
            constructor(props) {
                super(props)
            
                this.state = {
                    howis: "Parent",
                    name: "Readers"
                }
                this.changing = this.changing.bind(this)
            }
            
            changing(howis = "Parent"){
                this.setState(prevState=>({
                    howis: howis,
                    name: "learners"
                }))
            }
            render() {
                return (
                    <div>
                        <p>{this.state.howis} - {this.state.name}</p>
                        
                        <Child handle={this.changing}/>
                    </div>
                )
            }
        }
        Child Component
        function Child(props) {
            return (
                
                <div>
                    <button onClick={()=>props.handle("child")}>click child change parent<button>
                </div>
            )
        }
